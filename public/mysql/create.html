DDL: CREATE (banco e tabelas)

Exemplos de como criar bancos de dados e tabelas com constraints, chaves primárias e estrangeiras:

-- Criando um banco de dados com configuração de charset

CREATE DATABASE loja CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci;

USE loja;

-- Criando tabela de clientes com chave primária auto-incrementada

CREATE TABLE clientes (

cliente_id INT PRIMARY KEY AUTO_INCREMENT,

nome VARCHAR(100) NOT NULL,

email VARCHAR(150) UNIQUE,

telefone VARCHAR(20),

data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP

) ENGINE=InnoDB;

-- Tabela com chave estrangeira (Foreign Key)

CREATE TABLE pedidos (

pedido_id INT PRIMARY KEY AUTO_INCREMENT,

cliente_id INT NOT NULL,

total DECIMAL(10,2) NOT NULL DEFAULT 0,

status ENUM('novo', 'pendente', 'pago', 'enviado', 'entregue'),

data_criacao DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,

CONSTRAINT fk_pedidos_clientes

FOREIGN KEY (cliente_id) REFERENCES clientes(cliente_id)

ON UPDATE CASCADE ON DELETE RESTRICT

) ENGINE=InnoDB;

Dicas importantes:

  • Defina sempre o charset (UTF8MB4 é recomendado para suporte completo Unicode)
  • Use restrições (constraints) para garantir a integridade dos dados
  • Configure ações de ON UPDATE e ON DELETE nas chaves estrangeiras
  • Defina um engine apropriado (InnoDB para transações e integridade referencial)
