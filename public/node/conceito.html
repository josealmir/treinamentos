Exemplo de Rota

src/routes/users.ts TypeBox 
const Params = Type.Object({ id: Type.String() });

app.get('/users/:id', {
    schema: { params: Params }
}, async (req, reply) => {
    const { id } = req.params;
    return { id, name: 'Alice' };
});
