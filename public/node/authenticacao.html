import Fastify from 'fastify';
import jwt from '@fastify/jwt';

const app = Fastify();

app.register(jwt, {
  secret: process.env.JWT_SECRET,
  sign: { expiresIn: '24h' }
});

// Rota de login
app.post('/login', async (req, reply) => {
  const { email, password } = req.body;
  const token = await reply.jwtSign({ sub: email });
  return { token };
});
